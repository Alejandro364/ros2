FROM osrf/ros:humble-desktop

RUN apt-get update && apt-get install -y nano geany

# --- GUI vía navegador (noVNC) ---
RUN apt-get update && apt-get install -y \
    gedit \
    xvfb x11vnc novnc websockify \
    openbox xterm dbus-x11 x11-xserver-utils \
    && rm -rf /var/lib/apt/lists/*

# creamos un script de arranque .....
RUN printf '%s\n' \
'#!/usr/bin/env bash' \
'set -e' \
'export DISPLAY=:1' \
'Xvfb :1 -screen 0 1280x800x24 -ac +extension GLX +render -noreset &' \
'sleep 1' \
'mkdir -p /root/.config/autostart' \
'eval "$(dbus-launch --sh-syntax)"' \
'openbox &' \
'xsetroot -solid "#2e3440" || true' \
'x11vnc -display :1 -forever -shared -nopw -rfbport 5900 -xkb &' \
'websockify --web=/usr/share/novnc/ 6080 localhost:5900' \
> /root/start_novnc.sh && chmod +x /root/start_novnc.sh

# exponemos el puerto y arrancamos el servicio
EXPOSE 6080
CMD ["/root/start_novnc.sh"]
